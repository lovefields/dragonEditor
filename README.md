## Using
```js
const editor = dragonEditor.init('.editor_area', {
    'mediaUploadURL' : 'yourURL',
    'mediaUpdateURL' : 'yourURL',
    'mediaDelURL' : 'yourURL',
    'defaultLinkBoxImage' : 'imageURL'
});
```
